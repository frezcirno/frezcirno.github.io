送分题

```c++
class Solution {
public:
    char findTheDifference(string s, string t) {
        map<char, int>count;
        for(auto&&c:s){
            count[c]++;
        }
        for(auto&&c:t){
            count[c]--;
        }
        return find_if(count.begin(),count.end(),[&](auto it){return it.second!=0;})->first;
    }
};
```
