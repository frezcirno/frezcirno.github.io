找规律题

```c++
class Solution {
public:
    vector<bool> prefixesDivBy5(vector<int>& A) {
        vector<bool> ret;
        int res = 0;
        for(auto&&a:A){
            res = res * 2 + a;
            res %= 10;
            if(res % 5 == 0){
                ret.push_back(true);
            }else{
                ret.push_back(false);
            }
        }
        return ret;
    }
};
```
