双重拓扑排序: 组内 & 组间

拓扑排序
```c++
vector<int> topSort(vector<int>& deg, vector<vector<int>>& graph, vector<int>& items) {
    queue<int> Q;
    for (auto& item: items) {
        if (deg[item] == 0) {
            Q.push(item);
        }
    }
    vector<int> res;
    while (!Q.empty()) {
        int u = Q.front(); 
        Q.pop();
        res.emplace_back(u);
        for (auto& v: graph[u]) {
            if (--deg[v] == 0) {
                Q.push(v);
            }
        }
    }
    return res.size() == items.size() ? res : vector<int>{};
}
```
