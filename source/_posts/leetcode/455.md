```c++
class Solution {
public:
    int findContentChildren(vector<int>& g, vector<int>& s) {
        sort(g.begin(),g.end()); // 1 2 3
        sort(s.begin(),s.end()); // 1 1
        int count=0;
        for(int i=s.size()-1, gg=g.size()-1;i>=0&&gg>=0;i--,gg--,count++){
            int ss=s[i];
            while(gg>=0&&g[gg]>ss){
                gg--;
            }
            if(gg<0) count--;
        }
        return count;
    }
};
```
