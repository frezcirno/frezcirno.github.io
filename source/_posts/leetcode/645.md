/**
 * @param {number[]} nums
 * @return {number[]}
 */
var findErrorNums = function(nums) {
    let ex = new Array(nums.length);
    let dup,lost;
    for(num of nums){
        if (!ex[num-1]){
            ex[num-1]=1;
        }else{
            dup=num;
        }
    }
    for(let i=0;i<nums.length;i++){
        if(!ex[i]){
            lost=i+1;
            break;
        }
    }
    return [dup, lost]
};
