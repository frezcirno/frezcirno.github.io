## 题意

给定一些可以任意交换位置的组, 求整个字符串的最小字典序

## 思路

1. 用并查集把所有组关联在一起
2. 对于每个可交换组, 把所有候选字母依次填入

## 收获

1. unordered_map的使用
2. c++的for-range语法

```c++
for(auto&[i,group]:groups){
    sort(group.begin(),group.end(),greater());
}
```
3. 重温并查集

## 代码

参考官方题解优化完毕

```c++
class Solution {
public:
    vector<int>parents,rank;
    int dj_root(int i){
        return (parents[i]==i ? i : parents[i]=dj_root(parents[i]));
    }
    void dj_union(int a, int b){
        a=dj_root(a);
        b=dj_root(b);
        if(a==b)return;
        if (rank[a]<rank[b])swap(a,b);
        rank[a]+=rank[b];
        parents[a]=b;
    }
    string smallestStringWithSwaps(string s, vector<vector<int>>& pairs) {
        int n=s.size();
        parents.resize(n);
        rank.resize(n,1);
        for(int i=0;i<n;i++){
            parents[i]=i;
        }
        for(auto&&pair:pairs){
            dj_union(pair[0],pair[1]);
        }

        unordered_map<int,string>groups;
    
        for(int i=0;i<n;i++){
            int r=dj_root(i);
            groups[r].push_back(s[i]);
        }

        for(auto&[i,group]:groups){
            sort(group.begin(),group.end(),greater());
        }

        for(int i=0;i<n;i++){
            int g=dj_root(i);
            s[i]=groups[g].back();
            groups[g].pop_back();
        }

        return s;
    }
};
```
