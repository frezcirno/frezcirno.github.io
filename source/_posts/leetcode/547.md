无向图给出邻接表, 计算连通分量个数

```c++
class Solution {
public:
    int findCircleNum(vector<vector<int>>& isConnected) {
        int len = isConnected.size();
        int count = 0;
        vector<int>flag(len,0);
        for(int i=0;i<len;i++){
            if(!flag[i]){
                count++;
                queue<int>found;
                found.push(i);
                flag[i]=1;
                while(!found.empty()){
                    int cur = found.front(); found.pop();
                    for(int j=0;j<len;j++){
                        if(!flag[j]&&isConnected[cur][j]){
                            found.push(j);
                            flag[j]=1;
                        }
                    }
                }
            }
        }
        return count;
    }
};
```
