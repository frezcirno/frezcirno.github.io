寻找{(x,y)|x+y=m,x,y∈S}

```c++
#include <iostream>
#include <unordered_map>
using namespace std;

int main()
{
    int n, m;
    cin >> n >> m;
    unordered_map<int, int> S;
    for (int i = 0; i < n; i++) {
        int x;
        cin >> x;
        S[x]++;
    }

    int minx = 0;
    for (auto &&k : S) {
        int x = k.first;
        int y = m - x;
        if (x < y && S.find(y) != S.end() || x == y && S[x] > 1) {
            if (!minx || x < minx)
                minx = x;
        }
    }
    if (minx) {
        cout << minx << " " << m - minx << endl;
    } else {
        cout << "No Solution" << endl;
    }
    return 0;
}
```
